// A single user
export interface User {
  username: string;
  password: string;
}

export class Users {
  private users: User[] = [];

  registerUser(
    username: string,
    password: string,
  ): User | void {
    // Check username uniqueness
    for (const user of this.users) {
      if (user.username == username) {
        console.error("user with username already exists");
        return;
      }
    }
    const user: User = {
      username,
      password,
    };
    this.users.push(user);
    return user;
  }

  authenticateUser(
    username: string,
    password: string,
  ): void {
    for (const user of this.users) {
      if (user.username == username) {
        // Username match, password match
        if (user.password == password) {
          console.error("success");
          return;
        } // Username match, password mismatch
        else {
          console.error("wrong password, try again");
          return;
        }
      }
    }
    // Username doesn't exist
    console.error("user not found");
    return;
  }

  getUserName(user: User): string {
    return user.username;
  }

  getUsers(): User[] {
    return this.users;
  }

  getUsersString(): string[] {
    const users: string[] = [];
    for (const user of this.users) {
      users.push(user.username);
    }
    return users;
  }

  equals(user1: User, user2: User): boolean {
    return (user1.username == user2.username &&
      user1.password == user2.password);
  }
}
